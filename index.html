library(tm) 
library(rpart.plot)

df_labels = read.table('labels.csv',sep="|",header=T)
df_labels$FullText = as.character(df_labels$FullText)
docs <- Corpus(DataframeSource(as.data.frame(df_labels[,7])))   
docs <- tm_map(docs, PlainTextDocument)  
docs <- tm_map(docs, removeWords,c("the","name_res","for")) 
docs <- tm_map(docs, removeWords,stopwords()) 
dtm <- DocumentTermMatrix(docs)
tdm <- TermDocumentMatrix(docs)
dtm = removeSparseTerms(dtm, 0.98)
m = as.matrix(dtm)
df.classification = as.data.frame(m)
df.classification$Label = df_labels$Label

df.classification$Label[df.classification$Label != 'spam'] = 'service'

inxs = which(df.classification$Label == 'spam')
fit <- rpart(factor(Label) ~ ., data = df.classification)
inxs = df_labels$AutoID


prp(fit)
